📖 Usage Examples

How to implement these loaders in your project

HTML

<!-- Spinner Loader -->
<div class="spinner"></div>

<!-- Pulse Loader -->
<div class="pulse-loader">
    <div class="pulse-dot"></div>
    <div class="pulse-dot"></div>
    <div class="pulse-dot"></div>
</div>

<!-- Skeleton Loader -->
<div class="skeleton skeleton-text"></div>

JavaScript (Show/Hide)

// Show loader
function showLoader() {
    document.getElementById('loader').classList.add('active');
}

// Hide loader
function hideLoader() {
    document.getElementById('loader').classList.remove('active');
}

// Loader with timeout
function loadWithLoader() {
    showLoader();
    setTimeout(() => {
        hideLoader();
    }, 2000);
}

Full Page Loader

<div id="pageLoader" class="loader-overlay">
    <div class="spinner spinner-lg"></div>
</div>

<style>
.loader-overlay {
    position: fixed;
    top: 0;
    left: 0;
    right: 0;
    bottom: 0;

    backdrop-filter: blur(10px);

    opacity: 0;
    visibility: hidden;

</style>

Button Loader

<button class="btn-primary" onclick="submitForm(this)">
    <span class="btn-text">Submit</span>
    <span class="btn-loader hidden"><div class="spinner spinner-sm"></div></span>
</button>

<script>
function submitForm(btn) {
    const text = btn.querySelector('.btn-text');
    const loader = btn.querySelector('.btn-loader');
    text.classList.add('hidden');
    loader.classList.remove('hidden');
    
    setTimeout(() => {
        text.classList.remove('hidden');
        loader.classList.add('hidden');
    }, 2000);
}
</script>
